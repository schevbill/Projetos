GENERATOR E TRIGGER -->> FAZER ID SER AUTO INCREMENTADO

CREATE GENERATOR GEN_FUNCIONARIO_PK;
SET TERM ^;
CREATE TRIGGER TRI_FUNCIONARIO_PK FOR FUNCIONARIO
ACTIVE BEFORE INSERT POSITION 0
as
begin
  if (new.ID_FUNCIONARIO = 0) then
     new.ID_FUNCIONARIO=gen_id(GEN_FUNCIONARIO_PK,1);
end
 ^
COMMIT WORK ^
SET TERM ;^
