CREATE TABLE GRUPO(
ID_GRUPO INTEGER NOT NULL PRIMARY KEY,
DESCRICAO VARCHAR(60)
);


CREATE GENERATOR GEN_GRUPO_PK;
SET TERM ^;
CREATE TRIGGER TRI_GRUPO_PK FOR GRUPO
ACTIVE BEFORE INSERT POSITION 0
as
begin
  if (new.ID_GRUPO = 0) then
     new.ID_GRUPO=gen_id(GEN_GRUPO_PK,1);
end
 ^
COMMIT WORK ^
SET TERM ;^
